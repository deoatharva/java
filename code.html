//implementing bubble sort
#include<iostream.h>
#include<conio.h>

void get(int i, int n, int *la) {
    for(i = 0; i <= n; i++) {
        cout << "enter the " << i << " element :";
        cin >> la[i];
    }
}

void bubble(int n, int *la) {
    int pass, ch, ptr;
    for(pass = 0; pass < n; pass++) {
        for(ptr = 0; ptr < n - 1; ptr++) {
            if(la[ptr] > la[ptr + 1]) {
                ch = la[ptr];
                la[ptr] = la[ptr + 1];
                la[ptr + 1] = ch;
            }
        }
    }
}

void display(int i, int n, int *la) {
    for(i = 0; i <= n; i++) {
        cout << la[i] << endl;
    }
}
